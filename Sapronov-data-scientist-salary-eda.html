def plot_barplot(df, speciality_name, group_name):
    
    top_salary = df[(df['Data scientist or machine learning specialist']==1)]\
            .groupby(group_name).Salary.median().sort_values(ascending=False).reset_index()
    
    plt.figure(figsize = (12,10))

    ax = sns.barplot(x='Salary', y=group_name, data=top_salary)
    plt.title("   2020   Data Science", fontsize = 16, pad=10)
    ax.set_xlabel(", ..", fontsize=14, labelpad=5)
    ax.set_ylabel("")
    ax.tick_params(labelsize=12)

    plt.show()
